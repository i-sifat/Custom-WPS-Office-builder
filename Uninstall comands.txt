# Find the installed WPS package
dpkg -l | grep -i wps

# Purge all installed WPS-related packages
sudo apt purge $(dpkg -l | awk '/^ii/ && /wps/ {print $2}')

# Remove unused dependencies
sudo apt autoremove --purge

# Remove user configuration and cache
rm -rf ~/.config/Kingsoft \
       ~/.local/share/Kingsoft \
       ~/.cache/Kingsoft \
       ~/.fonts/wps-office

# Verify removal
which wps
dpkg -l | grep -i wps
